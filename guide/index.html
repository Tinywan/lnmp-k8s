# Kubectl

# run

$ kubectl run nginx \
    --image=nginx:1.17.3-alpine \
    --command -- cmd arg1 \
    -- arg1 arg2 \
    --port=80

# exec

$ kubectl exec -it nginx -- sh

# expose

$ kubectl expose deployment nginx --port=80 --target-port=80

# logs

$ kubectl logs POD_NAME
