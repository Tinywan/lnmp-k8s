# kubectl

# 增加 hosts

执行 kubectl 的主机必须能 ping 通节点,可以在主机增加 hosts

NODE_IP NODE_NAME

否则将不能执行 exec port-forward 等命令.

# run

$ kubectl run nginx \
    --image=nginx:1.17.7-alpine \
    --command -- cmd arg1 \
    -- arg1 arg2 \
    --port=80

# exec

$ kubectl exec -it nginx -- sh

# expose

$ kubectl expose deployment nginx --port=80 --target-port=80

# logs

$ kubectl logs POD_NAME
